SELECT BRANCH_ZIP as ZIPCODE,CONCAT('$',format_number(SUM(TRANSACTION_VALUE),2)) AS AMOUNT
FROM CDW_SAPP_CREDITCARD
JOIN CDW_SAPP_BRANCH ON 
CDW_SAPP_BRANCH.BRANCH_CODE = CDW_SAPP_CREDITCARD.BRANCH_CODE
GROUP BY BRANCH_ZIP
ORDER BY AMOUNT DESC
limit 20;



2)
SELECT TRANSACTION_TYPE, QUARTER,sum(TRANSACTION_VALUE)
from CDW_SAPP_CREDITCARD
JOIN CDW_SAPP_TIME ON 
CDW_SAPP_CREDITCARD.timeid = CDW_SAPP_TIME.timeid
WHERE CDW_SAPP_TIME.year = '2018'
GROUP BY  QUARTER, TRANSACTION_TYPE;
